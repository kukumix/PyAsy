PyAsy Examples¶

A very simple example¶

"""A simple example of how to use PyAsy."""

import numpy
import pyasy.plot

plot = pyasy.plot.Plot()

x = numpy.linspace(-10.0, 20.0, 101)
y = x**2

plot.line(x, y)
plot.axis(xlabel='$x$', ylabel='$y$')
plot.shipout('simple')

Contour example¶

"""A simple example of how to use PyAsy."""

import math
import numpy
import pyasy.plot

plot = pyasy.plot.Plot(size=(4,4,True))

x = numpy.linspace(-math.pi, math.pi, 201)
y = numpy.linspace(-math.pi, math.pi, 201)
z = numpy.zeros((x.size,y.size))
for i in xrange(x.size):
    for j in xrange(y.size):
        z[i,j] = numpy.sin(1.0/(1.0 + x[i]**2 + y[j]**2))

plot.density(x, y, z,
             bar={'label': '$z$',
                  'initial': (4.0, -math.pi),
                  'final': (4.25, math.pi)})
plot.axis(xlabel='$x$', ylabel='$y$')
plot.shipout('contour')
